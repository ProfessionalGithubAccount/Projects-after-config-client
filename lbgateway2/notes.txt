used this in application.yml in hellogateway..before adding routes or filters
spring:

  application:

    name: spring-cloud-gateway-app

  cloud:

    gateway:

      routes:

        - id: helloservice-route

          uri: http://localhost:8071

          predicates:

            - Path=/rest/service/**
management:
  endpoints:
    web:
      exposure:
        include: "*"
  endpoint:
    health:
      show-details: always
    gateway:
      enabled: true
          
server:
 port: 8093
